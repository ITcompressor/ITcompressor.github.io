Show tables

Create new user 
CREATE USER 'qwer'@'%' IDENTIFIED BY 'password';
Grant remote connection permissions 
GRANT ALL PRIVILEGES ON *.* TO 'root'@'%' IDENTIFIED BY 'password';
Grant all permissions 
GRANT GRANT OPTION ON *.* TO 'qwer'@'%';
GRANT ALL PRIVILEGES ON * . * TO 'qwer'@'%';
FLUSH PRIVILEGES;
