Установка докера (Docker installation) 
sudo apt-get remove docker docker-engine docker.io
sudo apt-get install apt-transport-https ca-certificates curl software-properties-common
curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo apt-key add -
sudo apt-key fingerprint 0EBFCD88
sudo add-apt-repository "deb [arch=amd64] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable"
sudo apt-get update
sudo apt-get install docker-ce
sudo docker run hello-world
sudo curl -L "https://github.com/docker/compose/releases/download/1.22.0/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
sudo chmod +x /usr/local/bin/docker-compose
Уловки для запуска докера не из под рута (run docker without root) 
sudo usermod -aG docker $USER
newgrp docker
sudo systemctl restart docker
sudo chmod 666 /var/run/docker.sock
Просмотр всех контейнеров (view all containers) 
docker ps -a
Просмотр запущенных контейнеров (view all running containers) 
docker ps
Запуск контейнера (start container) 
docker start $container_name
Просмотр информации о контейнере (view information about container) 
docker inspect $conainer_name
Собрать контейнер по Dockerfile (build dockerfile) 
docker build .
Просмотр лога контейнера (view container log) 
docker log $container_name
Удаление контейнера (remove container) 
docker rm $container_name
Создание контейнера через имейдж с форвардингом портов, где 8080 порт хоста, а 80 порт контейнера (run container with forward ports) 
docker run -d -p 8080:80 $container
Консоль к контейнеру (run a docker container's bash) 
docker exec -it $container_name bash
Пример запуска утилиты для докер контейнера (example run util in docker container) 
docker exec infra_mongodb_1 mongoexport -d devbot -c botLog -o /root/test.json
Удаление всех неактивных контейнеров (remove all inactive containers) 
docker rm -v $(docker ps -aq -f status=exited)
Создание собственного имеджа (create custom image) 
docker run -it --name myapp --hostname myapp ubuntu bash
apt update
apt install cowsay
docker commit myapp MY_DOCKER_ID/MY_IMAGE
docker push MY_DOCKER_ID/MY_IMAGE
