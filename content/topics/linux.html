Монтирование разделов

Посмотреть информацию о подключенных дисках 
sudo fstab -l
Редактирование автомонтирование разделов 
sudo nano /etc/fstab
Атомонтирование раздела ext4 
/dev/sda1 /home/qwer/DATA   ext4  defaults       0  0
Пример монтирования расшаренной виндой директорий 
nautilus smb://winda-pc/

Автозагрука скриптов

Пример команды для запуска в скрине 
screen -dmS $SCREEN_NAME /bin/bash -c "/opt/somescrip.sh 2>&1 > /opt/mylog.log"
Манипуляции для включения rc.local (актуально для новых версий убунту 18+) 
# touch /etc/rc.local
# chmod +x /etc/rc.local
добавляем это 
#!/bin/sh -e
#
# rc.local
#
# This script is executed at the end of each multiuser runlevel.
# Make sure that the script will "exit 0" on success or any other
# value on error.
#
# In order to enable or disable this script just change the execution
# bits.
#
# By default this script does nothing.

exit 0
аутивируем 
sudo systemctl enable rc-local
