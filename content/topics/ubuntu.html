Установка samba 
sudo apt install system-config-samba

sudo system-config-samba
Возможный фикс samba, если не находит нужного файла

Установка bash-completion для автокомплита в консоли 
apt install bash-completion
Настройки для .bashrc (если не работает автокомплит + цветовая схема терминала) 
if ! shopt -oq posix; then
  if [ -f /usr/share/bash-completion/bash_completion ]; then
    . /usr/share/bash-completion/bash_completion
  elif [ -f /etc/bash_completion ]; then
    . /etc/bash_completion
  fi
fi

export PS1="\[\033[36m\]\u\[\033[m\]@\[\033[32m\]\h:\[\033[33;1m\]\w\[\033[m\]\$ "
export CLICOLOR=1
export LSCOLORS=ExFxBxDxCxegedabagacad
alias ls='ls -GFh'

Манипуляции для включения rc.local (актуально для новых версий убунту 18+) 
# touch /etc/rc.local
# chmod +x /etc/rc.local
добавляем это 
#!/bin/sh -e
#
# rc.local
#
# This script is executed at the end of each multiuser runlevel.
# Make sure that the script will "exit 0" on success or any other
# value on error.
#
# In order to enable or disable this script just change the execution
# bits.
#
# By default this script does nothing.

exit 0
активируем 
sudo systemctl enable rc-local
