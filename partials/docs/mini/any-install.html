Установка LAMP (Linux Apache Mysql PHP) 
sudo apt-get install apache2 mysql-server php5 php5-mysql phpmyadmin
Пример установки библиотеки для maven 
mvn install:install-file -Dfile=/path_to_jar/sqljdbc4.jar -DgroupId=com.microsoft.sqlserver -DartifactId=sqljdbc4 -Dversion=4.0 -Dpackaging=jar
Установка gnome shell (графическая оболочка) 
sudo apt-get install gnome-shell gnome-session gnome-tweak-tool gdm
Установка net-tools (ifconfig) 
apt install net-tools
Установка jdk8 
wget repository.jelasticloud.com/scripts/jvm/8/jdk8 && chmod +x jdk8 && ./jdk8 && rm -rf jdk8
Установка jdk7 update 51 
wget repository.jelasticloud.com/scripts/jvm/7u51/jdk7u51 && chmod +x jdk7u51 && ./jdk7u51 && rm -rf jdk7u51
Установка sdkman для управления версиями утилит 
curl -s "https://get.sdkman.io" | bash
source "$HOME/.sdkman/bin/sdkman-init.sh"
